  <xsl:output method = "xml" indent = "yes" /> 

  <!-- Example of an XML criterion stream -->
  <!--  
<ParametersList>
    <Parameter Name = "Type" Type = "String" Alias = "Type" Shown = "Yes" Mandatory = "No" Sensitive = "No" AccessMode = "ReadWrite">
      <DefaultValue></DefaultValue>
    </Parameter>
    <Parameter Name = "Search string" Type = "String" Alias = "Search string" Shown = "Yes" Mandatory = "No" Sensitive = "No" AccessMode = "ReadWrite">
      <DefaultValue>Product_Prt</DefaultValue>
    </Parameter>
  </ParametersList>
 -->

  <!-- Define some internal XSL variables -->

  <xsl:variable name = "MySearchString">
     <xsl:value-of select = "//Parameter[@Name='Search string']/CurrentValue"/>
  </xsl:variable> 
  <xsl:variable name="length">
    <xsl:value-of select="string-length($MySearchString)" /> 
  </xsl:variable>

  <!-- Scan XML to rebuilt Parameter nodes inside the ParametersList node 
       NB: All under RenderingForm node could be ignored, it is automatically regenerated to define a layout for display -->
  
  <xsl:template match = "/"> 
    <xsl:element name="ParametersList"> 
      <xsl:apply-templates/> 
    </xsl:element> 
  </xsl:template> 

  <xsl:template match = "//MainFrame"> 
  </xsl:template> 

  <xsl:template match = "//Parameter"> 

    <!-- Define a set of discrete values to criterion parameter Type, define a default value and change Alias -->

    <xsl:choose>
      
        <!-- Customisation of the 'Type' parameter of the Parent Query-->
    
        <xsl:when test = "@Name='Type'" > 
          <xsl:element name="Parameter"> 
            <xsl:attribute name="Name">Object Type</xsl:attribute> 
            <xsl:attribute name="Type">String</xsl:attribute> 
            <xsl:attribute name="Alias">Object Type</xsl:attribute>
            <xsl:attribute name="Shown">Yes</xsl:attribute>
            <xsl:attribute name="Mandatory">Yes</xsl:attribute>
            <xsl:attribute name="Sensitive">No</xsl:attribute>
            <xsl:attribute name = "AccessMode">ReadWrite</xsl:attribute>
            <xsl:attribute name = "DefaultListExtensible">Yes</xsl:attribute>

            <!--  List of authorized values   --> 
            <xsl:element name="DefaultValuesList">
                <xsl:element name="DefaultValue">
                  <xsl:text>Product</xsl:text> 
                </xsl:element>
                <xsl:element name="DefaultValue">
                  <xsl:text>Part</xsl:text> 
                </xsl:element>
                <xsl:element name="DefaultValue">
                  <xsl:text>CATPart</xsl:text> 
                </xsl:element>
                <xsl:element name="DefaultValue">
                  <xsl:text>CATProduct</xsl:text> 
                </xsl:element>
                <xsl:element name="DefaultValue">
                  <xsl:text>Model</xsl:text> 
                </xsl:element>
            </xsl:element>
            <!--  Current value selected   --> 
            <xsl:element name="CurrentValue"><xsl:copy-of select="./CurrentValue"/> 
            </xsl:element>
          </xsl:element> 

        <!-- Additional parameter for Supplier Information : This is not mapped to any parameter of the standard query-->
          <xsl:element name="Parameter">
            <xsl:attribute name="Name">Supplier Name</xsl:attribute> 
            <xsl:attribute name="Type">String</xsl:attribute> 
            <xsl:attribute name="Alias">Supplier Name</xsl:attribute> 
            <xsl:attribute name="Shown">Yes</xsl:attribute> 
            <xsl:attribute name="Mandatory">No</xsl:attribute> 
            <xsl:attribute name="Sensitive">No</xsl:attribute> 
            <xsl:attribute name="AccessMode">Read</xsl:attribute> 
            
            <!--  Current value selected   --> 
            
            <xsl:element name="CurrentValue">
              <xsl:text>Supplier 1</xsl:text> 
            </xsl:element>
          
          </xsl:element>
        </xsl:when> 

        <!-- Customisation of the 'Search string' parameter of the Parent Query-->
        <xsl:when test="@Name='Search string'">      
          
          <xsl:element name="Parameter">
            <xsl:attribute name="Name">Prefix</xsl:attribute> 
            <xsl:attribute name="Type">String</xsl:attribute> 
            <xsl:attribute name="Alias">Prefix</xsl:attribute> 
            <xsl:attribute name="Shown">Yes</xsl:attribute> 
            <xsl:attribute name="Mandatory">No</xsl:attribute> 
            <xsl:attribute name="Sensitive">No</xsl:attribute> 
            <xsl:attribute name="AccessMode">
              <xsl:value-of select="./@AccessMode" /> 
            </xsl:attribute>
    
            <!--  Current value selected   --> 
            <xsl:element name="CurrentValue">
              <xsl:text>Rec_</xsl:text>
             </xsl:element>
             
          </xsl:element>

          <xsl:element name="Parameter">
            <xsl:attribute name="Name">Part ID</xsl:attribute> 
            <xsl:attribute name="Type">String</xsl:attribute> 
            <xsl:attribute name="Alias">Part ID</xsl:attribute> 
            <xsl:attribute name="Shown">Yes</xsl:attribute> 
            <xsl:attribute name="Mandatory">No</xsl:attribute> 
            <xsl:attribute name="Sensitive">No</xsl:attribute> 
            <xsl:attribute name="AccessMode">ReadWrite</xsl:attribute> 
          
            <!--  Current value selected   --> 
          
            <xsl:element name="CurrentValue">
              <xsl:value-of select="$MySearchString" /> 
            </xsl:element>
          
          </xsl:element>
          
        </xsl:when>

     <!-- Other parameters are copied  without any modification -->
            
      <!--xsl:otherwise > 
        <xsl:copy-of select = "." /> 
      </xsl:otherwise--> 
      
    </xsl:choose> 
  </xsl:template> 

</xsl:stylesheet>
